\begin{abstract}
Solving for expressions with trig functions
\end{abstract}

\begin{mydef}
A trigonometric equation is an equation that contains trig functions. A trig Identity is a trig equation that is true for all $\theta$
\end{mydef}

\subsection{Examples}
\begin{align*}
Identity:\sin^2+\cos^2 & = 1 & Equation:\sin(\theta) + \cos(\theta) & = 1
\end{align*}

\section{Basic Trig Equations}
They are in the form of $T(\theta) = c$ where $c$ is some constant.

\subsection{Examples}
Solve the following problems first for all value of $\theta$ and then all values where $\frac{3\pi}{2} \leq \theta \leq 4\pi$
\begin{enumerate}[(a)]
\item $\sin(\theta) = \frac{1}{2}$ 
\item $\cos(\theta) = - \frac{-\sqrt[]{2}}{2}$
\item $\tan(\theta) = 2$
\end{enumerate}

\section{Solving Trig Equations by Factoring}
Find all values of theta that solve the following problems.

\begin{enumerate}[(a)]
\item $2\cos^2(\theta) - 7\cos(\theta) + 3 = 0$
\item $5\sin(\theta)\cos(\theta) + 4\cos(\theta) = 0$
\end{enumerate}
